#+title: Actions


* Github: déployer les configs avec colmena

Pour info : les actions github sont des fichier ~.yml~ situées dans le repertoire ~.github/workflows~.


#+Property: Header-args :tangle ../.github/workflows/deployment.yml
#+BEGIN_SRC yaml
name: deploy nixos configurations

on:
  workflow_dispatch:
  push:
    branches:
      - main

jobs:
  deploy-nixos:
    runs-on: ubuntu-latest

    steps:
      - name: "checkout repo"
        uses: actions/checkout@v6
      - name: "instal nix"
        uses: cachix/install-nix-action@v31
        with:
          nix_path: nixpkgs=channel:nixos-26.05
      - run: nix-env -i colmena -f '<nixpkgs>'
      - name: "Tailscale"
        uses: tailscale/github-action@v4
        with:
          oauth-client-id: ${{ Tailscale.TS_OAUTH_CLIENT_ID }}
          oauth-secret: ${{ Tailscale.TS_OAUTH_SECRET }}
          tags: tag:testlab
          ping: xen.blue-edmontosaurus.ts.net
      - name: "Configure SSH"
        env:
          SSH_USER: ${{ secrets.SSH_USER }}
          SSH_KEY: ${{ secrets.SSH_KEY }}
          SSH_HOST: ${{ secrets.SSH_IP }}
          SSH_PORT: ${{ secrets.SSH_PORT }}
        run: |
          mkdir -p ~/.ssh/
          echo "$SSH_KEY" > ~/.ssh/external-machine.key
          chmod 600 ~/.ssh/external-machine.key
          cat >>~/.ssh/config <<END
          Host external-machine
            HostName $SSH_HOST
            User $SSH_USER
            Port $SSH_PORT
            IdentityFile ~/.ssh/external-machine.key
            StrictHostKeyChecking no
          END
          ssh-keyscan -H ${{ secrets.SSH_HOST }} >> ~/.ssh/known_hosts
      - name: "deployment"
        run: |
          colmena build --on @dom0
          colmena apply boot --on @dom0

#+END_SRC
